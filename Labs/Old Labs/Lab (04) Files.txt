void write(string s) {
  fstream file;
  file.open(s + ".txt", ios::out);
  file << "HELLO";
  file.close();
}
void read(string& s) {
  fstream file;
  file.open(s + ".txt", ios::in);
  file >> s;
  file.close();
}
